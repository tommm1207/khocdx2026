                    warehouseDataCache = data;
                    if (callback) callback(warehouseDataCache);

        function initializeWarehouseView(viewType) {
            const viewId = `${viewType}-view`;
            if (document.getElementById(viewId)?.dataset.initialized) return;

            const now = new Date(), year = now.getFullYear(), month = now.getMonth();
            const firstDay = new Date(year, month, 1), lastDay = new Date(year, month + 1, 0);
            const toYYYYMMDD = (date) => date.toISOString().split('T')[0];

            const fromDateInput = document.getElementById(`${viewType}-from-date`);
            const toDateInput = document.getElementById(`${viewType}-to-date`);
            fromDateInput.value = toYYYYMMDD(firstDay);
            toDateInput.value = toYYYYMMDD(lastDay);

            document.getElementById(`apply-${viewType}-filter`).addEventListener('click', () => renderWarehouseTable(viewType, warehouseDataCache));
            document.getElementById(`reset-${viewType}-filter`).addEventListener('click', () => {
                fromDateInput.value = toYYYYMMDD(firstDay);
                toDateInput.value = toYYYYMMDD(lastDay);
                document.getElementById(`${viewType}-material-filter`).selectedIndex = 0;
                document.getElementById(`${viewType}-group-filter`).selectedIndex = 0;
                document.getElementById(`${viewType}-warehouse-filter`).selectedIndex = 0;
                document.getElementById(`${viewType}-search-filter`).value = '';
                renderWarehouseTable(viewType, warehouseDataCache);
            });
            document.getElementById(`${viewType}-table-header`).addEventListener('click', (e) => handleSortClick(e, viewType));

            ensureWarehouseData(data => {
                populateWarehouseFilters(viewType, data);
                renderWarehouseTable(viewType, data);
            });

            if(document.getElementById(viewId)) document.getElementById(viewId).dataset.initialized = 'true';
        }
        function populateWarehouseFilters(viewType, data) {
            if (!data) return;
            const allItems = [...(data.incoming || []), ...(data.outgoing || [])];
            const materials = [...new Set(allItems.map(i => i['VatLieu']).filter(Boolean))];
            const groups = [...new Set(allItems.map(i => i['NhomVatLieu']).filter(Boolean))];
            const warehouses = [...new Set(allItems.map(i => i['TenKho']).filter(Boolean))];

            const materialSelect = document.getElementById(`${viewType}-material-filter`);
            const groupSelect = document.getElementById(`${viewType}-group-filter`);
            const warehouseSelect = document.getElementById(`${viewType}-warehouse-filter`);

            const populate = (select, options) => {
                if (!select) return;
                while (select.options.length > 1) select.remove(1);
                options.forEach(o => select.add(new Option(o, o)));
            };

            populate(materialSelect, materials);
            populate(groupSelect, groups);
            populate(warehouseSelect, warehouses);
        }
        function renderWarehouseTable(viewType, data) {
            if (!data) return;
            const sourceData = viewType === 'stock-in' ? data.incoming : data.outgoing;
            if (!Array.isArray(sourceData)) {
                document.getElementById(`${viewType}-table-body`).innerHTML = `<tr><td colspan="11">Không có dữ liệu.</td></tr>`;
                return;
            }

            const fromDate = document.getElementById(`${viewType}-from-date`).value;
            const toDate = document.getElementById(`${viewType}-to-date`).value;
            const material = document.getElementById(`${viewType}-material-filter`).value;
            const group = document.getElementById(`${viewType}-group-filter`).value;
            const warehouse = document.getElementById(`${viewType}-warehouse-filter`).value;
            const search = document.getElementById(`${viewType}-search-filter`).value.toLowerCase();

            let filteredData = sourceData.filter(item => {
                const itemDate = item.Ngay ? item.Ngay.split('T')[0] : null;
                const searchMatch = !search || (item.DienGiai || '').toLowerCase().includes(search) || (item.VatLieu || '').toLowerCase().includes(search);
                return (!fromDate || (itemDate && itemDate >= fromDate)) &&
                       (!toDate || (itemDate && itemDate <= toDate)) &&
                       (!material || item.VatLieu === material) &&
                       (!group || item.NhomVatLieu === group) &&
                       (!warehouse || item.TenKho === warehouse) &&
                       searchMatch;
            });

            if (warehouseSortState.column) {
                filteredData.sort((a, b) => {
                    let valA = a[warehouseSortState.column];
                    let valB = b[warehouseSortState.column];
                    let comparison = 0;
                    const numericColumns = ['SoLuong', 'DonGia', 'ThanhTien'];
                    if (numericColumns.includes(warehouseSortState.column)) {
                        comparison = (parseFloat(valA) || 0) - (parseFloat(valB) || 0);
                    } else if (warehouseSortState.column === 'Ngay') {
                        comparison = (new Date(valA).getTime() || 0) - (new Date(valB).getTime() || 0);
                    } else {
                        comparison = (valA || '').toString().localeCompare((valB || '').toString(), 'vi');
                    }
                    return warehouseSortState.direction === 'asc' ? comparison : -comparison;
                });
            }
            
            const tbody = document.getElementById(`${viewType}-table-body`);
            if (tbody) {
                tbody.innerHTML = filteredData.map((item, index) => `
                    <tr>
                        <td>${index + 1}</td>
                        <td>${item.VatLieu || ''}</td>
                        <td>${item.NhomVatLieu || ''}</td>
                        <td>${formatDate(item.Ngay)}</td>
                        <td class="text-end">${item.SoLuong || 0}</td>
                        <td>${item.DonViTinh || ''}</td>
                        <td class="text-end">${formatCurrency(item.DonGia)}</td>
                        <td class="text-end">${formatCurrency(item.ThanhTien)}</td>
                        <td>${item.DienGiai || ''}</td>
                        <td>${item.TenKho || ''}</td>
                        <td>${item.HinhAnh ? `<img src="${item.HinhAnh}" alt="Ảnh" style="width: 50px; height: 50px; object-fit: cover;">` : ''}</td>
                    </tr>
                `).join('');
            }
            updateSortIcons(`${viewType}-view`, warehouseSortState);
        }
        function handleSortClick(e, viewType) {
            let sortState, renderFunc, dataCache;
            if (viewType === 'shopping') {
                sortState = shoppingSortState;
                dataCache = shoppingDataCache;
                renderFunc = () => renderShoppingTable(dataCache);
            } else { // 'stock-in' or 'stock-out'
                sortState = warehouseSortState;
                dataCache = warehouseDataCache;
                renderFunc = () => renderWarehouseTable(viewType, dataCache);
            }
            const header = e.target.closest('.sortable-header');
            if (!header) return;
            const columnKey = header.dataset.sortKey;
            if (sortState.column === columnKey) {
                sortState.direction = sortState.direction === 'asc' ? 'desc' : 'asc';
            } else {
                sortState.column = columnKey;
                sortState.direction = 'asc';
            }
            renderFunc();
        }
        function updateSortIcons(viewId, sortState) {
            document.querySelectorAll(`#${viewId} .sortable-header`).forEach(header => {
                const icon = header.querySelector('.sort-icon');
                if (icon) {
                    icon.className = 'fas sort-icon';
                    if (header.dataset.sortKey === sortState.column) {
                        icon.classList.add(sortState.direction === 'asc' ? 'fa-sort-up' : 'fa-sort-down', 'active');
                    } else {
                        icon.classList.add('fa-sort');
                    }
                }
            });
        }
