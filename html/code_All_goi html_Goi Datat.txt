    <script>
        const resultArea = document.getElementById('result');
        const loader = document.getElementById('loader');

        function runApiCall(action, params = {}) {
            showLoader(true);
            resultArea.textContent = 'Đang xử lý yêu cầu...';
            
            // Dòng này sử dụng scriptlet của Apps Script để lấy URL động của Web App
            const webAppUrl = "<?= ScriptApp.getService().getUrl() ?>";
            
            const queryParams = new URLSearchParams({ action, ...params });
            const finalUrl = `${webAppUrl}?${queryParams}`;

            fetch(finalUrl)
                .then(response => {
                    if (!response.ok) {
                        throw new Error(`Lỗi mạng: ${response.status} ${response.statusText}`);
                    }
                    return response.json();
                })
                .then(data => {
                    // Định dạng lại JSON để hiển thị đẹp hơn
                    resultArea.textContent = JSON.stringify(data, null, 2);
                    showLoader(false);
                })
                .catch(error => {
                    console.error('Lỗi khi gọi API:', error);
                    resultArea.textContent = `Đã xảy ra lỗi:\n\n${error.message}\n\nXem chi tiết trong Console (F12).`;
                    showLoader(false);
                });
        }

        function showLoader(isLoading) {
            loader.style.display = isLoading ? 'block' : 'none';
        }

        // Tự động chạy lấy thống kê khi tải trang lần đầu
        document.addEventListener('DOMContentLoaded', () => {
            runApiCall('getStats');
        });
    </script>
